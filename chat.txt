    startGame();

    // Asegurarse de que el estado inicial sea resoluble
    if (!isSolvable(initialTiles)) {
      // Si no es resoluble, intercambiar dos tiles
      [initialTiles[0], initialTiles[1]] = [initialTiles[1], initialTiles[0]];
    }
    setTiles(initialTiles);
    setEmptyTile(initialTiles.indexOf(totalTiles - 1));
    setMoves(0);
    setIsCompleted(false);
  };

  const shuffle = (array) => {
    let shuffled = array.slice();
    for (let i = shuffled.length - 1; i > 0; i--) {
      const j = Math.floor(Math.random() * (i + 1));
      [shuffled[i], shuffled[j]] = [shuffled[j], shuffled[i]];
    }
    return shuffled;
  };

  const isSolvable = (array) => {
    let inversionCount = 0;
    for (let i = 0; i < array.length - 1; i++) {
      for (let j = i + 1; j < array.length; j++) {
        if (array[i] > array[j]) inversionCount++;
      }
    }
    return inversionCount % 2 === 0;
  };

  const handleTileClick = (index) => {
    if (isAdjacent(index, emptyTile)) {
      let newTiles = tiles.slice();
      [newTiles[index], newTiles[emptyTile]] = [newTiles[emptyTile], newTiles[index]];
      setTiles(newTiles);
      setEmptyTile(index);
      setMoves(moves + 1);
      if (checkCompletion(newTiles)) {
        setIsCompleted(true);
      }
    }
  };

  const isAdjacent = (index1, index2) => {
    const row1 = Math.floor(index1 / gridSize);
    const col1 = index1 % gridSize;
    const row2 = Math.floor(index2 / gridSize);
    const col2 = index2 % gridSize;
    return (
      (Math.abs(row1 - row2) === 1 && col1 === col2) ||
      (Math.abs(col1 - col2) === 1 && row1 === row2)
    );
  };

  const checkCompletion = (currentTiles) => {
    for (let i = 0; i < totalTiles; i++) {
      if (currentTiles[i] !== i) return false;
    }
    return true;

  const tileSize = 300 / gridSize; // Tamaño de la imagen en px

      {/* Campo de estrellas */}
      <Starfield />

      {/* Overlay */}
      <div className="absolute inset-0 bg-black bg-opacity-60"></div>

      {/* Contenido */}
      <div className="z-10 text-center space-y-8 px-4">
        {/* Título */}
        <motion.h1
          initial={{ y: -100, opacity: 0 }}
          animate={{ y: 0, opacity: 1 }}
          transition={{ duration: 1, ease: 'backOut' }}
          className="text-5xl md:text-6xl font-extrabold text-white"
          style={{ fontFamily: "'Orbitron', sans-serif" }}
        >
          Puzzle de Exoplanetas
        </motion.h1>

        {/* Tablero del Puzzle */}
        <div
          className="relative mx-auto"
          style={{
            width: `${tileSize * gridSize}px`,
            height: `${tileSize * gridSize}px`,
            border: '4px solid white',
            position: 'relative',
          }}
        >
          {tiles.map((tile, index) => {
            if (tile === totalTiles - 1) return null; // Es el espacio vacío
            const row = Math.floor(tile / gridSize);
            const col = tile % gridSize;
            return (
              <motion.div
                key={index}
                className="absolute cursor-pointer border border-black"
                style={{
                  width: `${tileSize}px`,
                  height: `${tileSize}px`,
                  backgroundImage: `url(${IMAGE_URL})`,
                  backgroundSize: `${tileSize * gridSize}px ${tileSize * gridSize}px`,
                  backgroundPosition: `-${col * tileSize}px -${row * tileSize}px`,
                  top: `${Math.floor(index / gridSize) * tileSize}px`,
                  left: `${(index % gridSize) * tileSize}px`,
                }}
                onClick={() => handleTileClick(index)}
                whileHover={{ scale: 1.05 }}
                whileTap={{ scale: 0.95 }}
                transition={{ duration: 0.2 }}
              />
            );
          })}
        </div>

        {/* Botones de Control */}
        <div className="flex justify-center space-x-6">
          <button
            onClick={startGame}
            className="flex items-center space-x-2 text-white bg-transparent border-2 border-white px-4 py-2 rounded hover:bg-white hover:text-black transition"
          >
            <RefreshCcw className="w-5 h-5" />
            <span>Reiniciar</span>
          </button>
        </div>

        {/* Movimientos y Mensaje de Completado */}
        <div className="text-white">
          <p className="text-lg">Movimientos: {moves}</p>
          {isCompleted && (
            <motion.p
              initial={{ scale: 0 }}
              animate={{ scale: 1 }}
              transition={{ duration: 0.5 }}
              className="mt-4 text-2xl font-bold text-green-400"
            >
              ¡Has completado el puzzle!
            </motion.p>
          )}
        </div>

        {/* Navegación */}
        <div className="flex justify-center space-x-8 mt-8">

function Starfield() {
  const [stars, setStars] = useState([]);

  useEffect(() => {
    const generateStars = () => {
      return Array.from({ length: 200 }, () => ({
        x: Math.random() * 100,
        y: Math.random() * 100,

        opacity: Math.random() * 0.7 + 0.3,
        twinkleSpeed: Math.random() * 5 + 1,
      }));
    };
    setStars(generateStars());
  }, []);

  return (
    <>
      {stars.map((star, index) => (
        <motion.div
          key={index}
          className="absolute bg-white rounded-full star"
          style={{
            left: `${star.x}%`,
            top: `${star.y}%`,
            width: `${star.size}px`,

            opacity: star.opacity,
          }}
          animate={{ opacity: [0.3, star.opacity, 0.3] }}
          transition={{
            duration: star.twinkleSpeed,
            repeat: Infinity,
            ease: 'easeInOut',
          }}
        />
      ))}
    </>
  );
}

### Explicación Detallada

1. **Estado y Lógica del Juego**:
    - **`tiles`**: Representa el estado actual de las piezas del puzzle.
    - **`emptyTile`**: Índice de la pieza vacía.
    - **`moves`**: Cuenta los movimientos realizados por el jugador.
    - **`isCompleted`**: Indica si el puzzle ha sido completado.

2. **Inicialización del Juego**:
    - Se genera un arreglo de piezas numeradas de 0 a 8 (para un puzzle de 3x3).
    - Se realiza una mezcla aleatoria de las piezas asegurando que el estado inicial sea resoluble (paridad de inversiones).
    - Se reinician los contadores y se prepara el estado para jugar.

3. **Interacción del Usuario**:
    - Al hacer clic en una pieza, se verifica si está adyacente a la pieza vacía.
    - Si es así, se intercambian las posiciones y se actualiza el estado.
    - Se incrementa el contador de movimientos y se verifica si el puzzle está completo.

4. **Verificación de Compleción**:
    - Se comprueba si el arreglo de piezas está en orden secuencial, lo que indica que el jugador ha completado el puzzle.

5. **Renderizado de las Piezas del Puzzle**:
    - Cada pieza del puzzle muestra una parte de la imagen general mediante `background-position`.
    - Se utiliza `framer-motion` para animaciones al interactuar con las piezas.

6. **Fondo y Estrellas**:
    - Se reutiliza el efecto de fondo con estrellas usando el componente `Starfield` para mantener la coherencia estética con tu código original.

7. **Controles y Navegación**:
    - Botón para reiniciar el juego, estilizado y con iconos.
    - Iconos de navegación similares a los de tu código original, manteniendo la experiencia de usuario consistente.

8. **Estilos y Animaciones**:
    - Se utilizan clases de Tailwind CSS para estilos rápidos y responsivos.
    - `framer-motion` se utiliza para agregar animaciones suaves y atractivas.

### Consideraciones Adicionales

- **Responsividad**: El tamaño del puzzle puede adaptarse para diferentes tamaños de pantalla ajustando el valor de `tileSize` o utilizando unidades relativas.
- **Mejoras Futuras**:
    - **Selección de Imágenes**: Permitir al usuario escoger diferentes imágenes para el puzzle.

### **Detalles de las Nuevas Funcionalidades**

1. **Etiqueta Roja de Notificación en la Pestaña "Desafíos"**:
   - **Descripción**: Si hay al menos un desafío pendiente, se muestra una etiqueta roja en la pestaña "Desafíos" indicando el número de desafíos pendientes.
   - **Implementación**:
     ```javascript
     {tab === 'desafíos' && pendingChallenges > 0 && (
       <span className="absolute -top-2 -right-2 bg-red-500 text-white text-xs w-5 h-5 flex items-center justify-center rounded-full">
         {pendingChallenges}
       </span>
     )}
     ```
     - **Explicación**: Dentro del mapa de pestañas, se verifica si la pestaña actual es "desafíos" y si hay desafíos pendientes (`pendingChallenges > 0`). Si ambas condiciones se cumplen, se renderiza una etiqueta roja con el número de desafíos pendientes.

2. **Completar Desafíos con Animaciones Satisfactorias**:
   - **Descripción**: Al hacer clic en el botón "Completar" de un desafío, se actualiza su estado a completado y se muestra una animación que indica su finalización.
   - **Implementación**:
     - **Estado y Función para Completar Desafíos**:
       ```javascript
       const [dailyChallenges, setDailyChallenges] = useState([
         { name: "Exploración Rápida", description: "Visita 3 nuevos planetas hoy", reward: "50 XP", completed: false },
         { name: "Diplomacia Diaria", description: "Interactúa con 2 especies alienígenas", reward: "30 XP", completed: true },
         { name: "Estudio Estelar", description: "Analiza 5 estrellas diferentes", reward: "40 XP", completed: false }
       ])

       const completeChallenge = (challengeName) => {
         setDailyChallenges(prevChallenges => 
           prevChallenges.map(challenge => 
             challenge.name === challengeName ? { ...challenge, completed: true } : challenge
           )
         )
       }
       ```
       - **Explicación**: `completeChallenge` recibe el nombre del desafío y actualiza su estado a completado.

     - **Animación al Completar un Desafío**:
       ```javascript
       {challenge.completed ? (
         <motion.div
           initial={{ scale: 0 }}
           animate={{ scale: 1 }}
           transition={{ duration: 0.3 }}
           className="flex items-center text-green-400"
         >
           <CheckCircle className="w-6 h-6 mr-1" />
           <span className="font-semibold">Completado</span>
         </motion.div>
       ) : (
         <button
           onClick={() => completeChallenge(challenge.name)}
           className="flex items-center px-4 py-2 bg-blue-500 text-white rounded-full hover:bg-blue-600 transition-colors"
         >
           Completar
           <motion.div
             whileTap={{ scale: 0.9 }}
             className="ml-2"
           >
             <Zap className="w-5 h-5 inline" />
           </motion.div>
         </button>
       )}
       ```
       - **Explicación**: Si un desafío está completado, se muestra un ícono de check (`CheckCircle`) con el texto "Completado". De lo contrario, se muestra el botón "Completar". Al hacer clic en este botón, se completa el desafío y se dispara una animación de escalado utilizando `framer-motion`.

3. **Cambiar el Estado Visual del Desafío a "Completado"**:
   - **Descripción**: Una vez que un desafío se completa, su apariencia cambia para reflejar su nuevo estado.
   - **Implementación**:
     ```javascript
     className={`p-4 rounded-lg flex items-center ${
       challenge.completed ? 'bg-green-600 bg-opacity-20' : 'bg-white bg-opacity-10'
     }`}
     ```
     - **Explicación**: Si el desafío está completado, el fondo cambia a un tono verde con opacidad (`bg-green-600 bg-opacity-20`). Si no está completado, mantiene el fondo blanco semitransparente (`bg-white bg-opacity-10`).
